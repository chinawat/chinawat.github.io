\begin{eqnarray}
\frac{n(n+1)}{2}+(n+1)&=&\frac{n(n+1)+2(n+1)}{2}\\
&=&\frac{(n+1)[n+2]}{2}\\
\sum_{i=1}^{n+1}{i}&=&\frac{(n+1)(n+2)}{2}.
\end{eqnarray}
